139 static inline vui16_t
140 vec_clzh (vui16_t vra)
141 {
142  vui16_t r;
143 #ifdef _ARCH_PWR8
144 #ifndef vec_vclzh
145  __asm__(
146  "vclzh %0,%1;"
147  : "=v" (r)
148  : "v" (vra)
149  : );
150 #else
151  r = vec_vclzh (vra);
152 #endif
153 #else
154 //#warning Implememention pre power8
155  vui16_t n, nt, y, x, s, m;
156  vui16_t z= { 0,0,0,0, 0,0,0,0};
157  vui16_t one = { 1,1,1,1, 1,1,1,1};
158 
159  /* n = 16 s = 8 */
160  s = vec_splat_u16(8);
161  n = vec_add (s, s);
162  x = vra;
163 
164  /* y=x>>8 if (y!=0) (n=n-8 x=y) */
165  y = vec_sr(x, s);
166  nt = vec_sub(n,s);
167  m = (vui16_t)vec_cmpgt(y, z);
168  s = vec_sr(s,one);
169  x = vec_sel (x , y, m);
170  n = vec_sel (n , nt, m);
171 
172  /* y=x>>4 if (y!=0) (n=n-4 x=y) */
173  y = vec_sr(x, s);
174  nt = vec_sub(n,s);
175  m = (vui16_t)vec_cmpgt(y, z);
176  s = vec_sr(s,one);
177  x = vec_sel (x , y, m);
178  n = vec_sel (n , nt, m);
179 
180  /* y=x>>2 if (y!=0) (n=n-2 x=y) */
181  y = vec_sr(x, s);
182  nt = vec_sub(n,s);
183  m = (vui16_t)vec_cmpgt(y, z);
184  s = vec_sr(s,one);
185  x = vec_sel (x , y, m);
186  n = vec_sel (n , nt, m);
187 
188  /* y=x>>1 if (y!=0) return (n=n-2) */
189  y = vec_sr(x, s);
190  nt = vec_sub(n,s);
191  nt = vec_sub(nt,s);
192  m = (vui16_t)vec_cmpgt(y, z);
193  n = vec_sel (n , nt, m);
194 
195  /* else return (x-n) */
196  nt = vec_sub (n, x);
197  r = vec_sel (nt , n, m);
198 #endif
199 
200  return (r);
201 }

420 static inline vui16_t
421 vec_popcnth (vui16_t vra)
422 {
423  vui16_t r;
424 #ifdef _ARCH_PWR8
425 #ifndef vec_vpopcnth
426  __asm__(
427  "vpopcnth %0,%1;"
428  : "=v" (r)
429  : "v" (vra)
430  : );
431 #else
432  r = vec_vpopcnth (vra);
433 #endif
434 #else
435  //#warning Implememention pre power8
436  __vector unsigned short n, x1, x2, x, s;
437  __vector unsigned short ones = { 1,1,1,1, 1,1,1,1};
438  __vector unsigned short fives =
439  {0x5555,0x5555,0x5555,0x5555, 0x5555,0x5555,0x5555,0x5555};
440  __vector unsigned short threes =
441  {0x3333,0x3333,0x3333,0x3333, 0x3333,0x3333,0x3333,0x3333};
442  __vector unsigned short fs =
443  {0x0f0f,0x0f0f,0x0f0f,0x0f0f, 0x0f0f,0x0f0f,0x0f0f,0x0f0f};
444  /* n = 8 s = 4 */
445  s = ones;
446  x = vra;
447 
448  /* x = x - ((x >> 1) & 0x5555) */
449  x2 = vec_and (vec_sr (x, s), fives);
450  n = vec_sub (x, x2);
451  s = vec_add (s, s);
452 
453  /* x = (x & 0x3333) + ((x & 0xcccc) >> 2) */
454  x1 = vec_and (n, threes);
455  x2 = vec_andc (n, threes);
456  n = vec_add (x1, vec_sr (x2, s));
457  s = vec_add (s, s);
458 
459  /* x = (x + (x >> 4)) & 0x0f0f) */
460  x1 = vec_add (n, vec_sr (n, s));
461  n = vec_and (x1, fs);
462  s = vec_add (s, s);
463 
464  /* This avoids the extra load const. */
465  /* x = (x + (x << 8)) >> 8) */
466  x1 = vec_add (n, vec_sl (n, s));
467  r = vec_sr (x1, s);
468 #endif
469  return (r);
470 }

536 static inline vui16_t
537 vec_slhi (vui16_t vra, const unsigned int shb)
538 {
539  vui16_t lshift;
540  vui16_t result;
541 
542  if (shb < 16)
543  {
544  /* Load the shift const in a vector. The element shifts require
545  a shift amount for each element. For the immediate form the
546  shift constant is splatted to all elements of the
547  shift control. */
548  if (__builtin_constant_p(shb))
549  lshift = (vui16_t) vec_splat_s16(shb);
550  else
551  lshift = vec_splats ((unsigned short) shb);
552 
553  /* Vector Shift right bytes based on the lower 4-bits of
554  corresponding element of lshift. */
555  result = vec_vslh (vra, lshift);
556  }
557  else
558  { /* shifts greater then 15 bits return zeros. */
559  result = vec_xor ((vui16_t) vra, (vui16_t) vra);
560  }
561 
562  return (vui16_t) result;
563 }
