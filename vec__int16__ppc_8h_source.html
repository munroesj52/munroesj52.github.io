88 vec_clzh (vui16_t vra)
89 {
90  vui16_t r;
91 #ifdef _ARCH_PWR8
92 #ifndef vec_vclzh
93  __asm__(
94  "vclzh %0,%1;"
95  : "=v" (r)
96  : "v" (vra)
97  : );
98 #else
99  r = vec_vclzh (vra);
100 #endif
101 #else
102 //#warning Implememention pre power8
103  vui16_t n, nt, y, x, s, m;
104  vui16_t z= { 0,0,0,0, 0,0,0,0};
105  vui16_t one = { 1,1,1,1, 1,1,1,1};
106 
107  /* n = 16 s = 8 */
108  s = vec_splat_u16(8);
109  n = vec_add (s, s);
110  x = vra;
111 
112  /* y=x>>8 if (y!=0) (n=n-8 x=y) */
113  y = vec_sr(x, s);
114  nt = vec_sub(n,s);
115  m = (vui16_t)vec_cmpgt(y, z);
116  s = vec_sr(s,one);
117  x = vec_sel (x , y, m);
118  n = vec_sel (n , nt, m);
119 
120  /* y=x>>4 if (y!=0) (n=n-4 x=y) */
121  y = vec_sr(x, s);
122  nt = vec_sub(n,s);
123  m = (vui16_t)vec_cmpgt(y, z);
124  s = vec_sr(s,one);
125  x = vec_sel (x , y, m);
126  n = vec_sel (n , nt, m);
127 
128  /* y=x>>2 if (y!=0) (n=n-2 x=y) */
129  y = vec_sr(x, s);
130  nt = vec_sub(n,s);
131  m = (vui16_t)vec_cmpgt(y, z);
132  s = vec_sr(s,one);
133  x = vec_sel (x , y, m);
134  n = vec_sel (n , nt, m);
135 
136  /* y=x>>1 if (y!=0) return (n=n-2) */
137  y = vec_sr(x, s);
138  nt = vec_sub(n,s);
139  nt = vec_sub(nt,s);
140  m = (vui16_t)vec_cmpgt(y, z);
141  n = vec_sel (n , nt, m);
142 
143  /* else return (x-n) */
144  nt = vec_sub (n, x);
145  r = vec_sel (nt , n, m);
146 #endif
147 
148  return (r);
149 }

172 {
173  vui16_t r;
174 #ifdef _ARCH_PWR8
175 #ifndef vec_vpopcnth
176  __asm__(
177  "vpopcnth %0,%1;"
178  : "=v" (r)
179  : "v" (vra)
180  : );
181 #else
182  r = vec_vpopcnth (vra);
183 #endif
184 #else
185  //#warning Implememention pre power8
186  __vector unsigned short n, x1, x2, x, s;
187  __vector unsigned short ones = { 1,1,1,1, 1,1,1,1};
188  __vector unsigned short fives =
189  {0x5555,0x5555,0x5555,0x5555, 0x5555,0x5555,0x5555,0x5555};
190  __vector unsigned short threes =
191  {0x3333,0x3333,0x3333,0x3333, 0x3333,0x3333,0x3333,0x3333};
192  __vector unsigned short fs =
193  {0x0f0f,0x0f0f,0x0f0f,0x0f0f, 0x0f0f,0x0f0f,0x0f0f,0x0f0f};
194  /* n = 8 s = 4 */
195  s = ones;
196  x = vra;
197 
198  /* x = x - ((x >> 1) & 0x5555) */
199  x2 = vec_and (vec_sr (x, s), fives);
200  n = vec_sub (x, x2);
201  s = vec_add (s, s);
202 
203  /* x = (x & 0x3333) + ((x & 0xcccc) >> 2) */
204  x1 = vec_and (n, threes);
205  x2 = vec_andc (n, threes);
206  n = vec_add (x1, vec_sr (x2, s));
207  s = vec_add (s, s);
208 
209  /* x = (x + (x >> 4)) & 0x0f0f) */
210  x1 = vec_add (n, vec_sr (n, s));
211  n = vec_and (x1, fs);
212  s = vec_add (s, s);
213 
214  /* This avoids the extra load const. */
215  /* x = (x + (x << 8)) >> 8) */
216  x1 = vec_add (n, vec_sl (n, s));
217  r = vec_sr (x1, s);
218 #endif
219  return (r);
220 }

276 static inline vui16_t
277 vec_slhi (vui16_t vra, const unsigned int shb)
278 {
279  vui16_t lshift;
280  vui16_t result;
281 
282  if (shb < 16)
283  {
284  /* Load the shift const in a vector. The element shifts require
285  a shift amount for each element. For the immediate form the
286  shift constant is splatted to all elements of the
287  shift control. */
288  if (__builtin_constant_p(shb))
289  lshift = (vui16_t) vec_splat_s16(shb);
290  else
291  lshift = vec_splats ((unsigned short) shb);
292 
293  /* Vector Shift right bytes based on the lower 4-bits of
294  corresponding element of lshift. */
295  result = vec_vslh (vra, lshift);
296  }
297  else
298  { /* shifts greater then 15 bits return zeros. */
299  result = vec_xor ((vui16_t) vra, (vui16_t) vra);
300  }
301 
302  return (vui16_t) result;
303 }
