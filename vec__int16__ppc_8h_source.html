546 static inline vui16_t
547 vec_clzh (vui16_t vra)
548 {
549  vui16_t r;
550 #ifdef _ARCH_PWR8
551 #ifndef vec_vclzh
552  __asm__(
553  "vclzh %0,%1;"
554  : "=v" (r)
555  : "v" (vra)
556  : );
557 #else
558  r = vec_vclzh (vra);
559 #endif
560 #else
561 //#warning Implememention pre power8
562  vui16_t n, nt, y, x, s, m;
563  vui16_t z= { 0,0,0,0, 0,0,0,0};
564  vui16_t one = { 1,1,1,1, 1,1,1,1};
565 
566  /* n = 16 s = 8 */
567  s = vec_splat_u16(8);
568  n = vec_add (s, s);
569  x = vra;
570 
571  /* y=x>>8 if (y!=0) (n=n-8 x=y) */
572  y = vec_sr(x, s);
573  nt = vec_sub(n,s);
574  m = (vui16_t)vec_cmpgt(y, z);
575  s = vec_sr(s,one);
576  x = vec_sel (x , y, m);
577  n = vec_sel (n , nt, m);
578 
579  /* y=x>>4 if (y!=0) (n=n-4 x=y) */
580  y = vec_sr(x, s);
581  nt = vec_sub(n,s);
582  m = (vui16_t)vec_cmpgt(y, z);
583  s = vec_sr(s,one);
584  x = vec_sel (x , y, m);
585  n = vec_sel (n , nt, m);
586 
587  /* y=x>>2 if (y!=0) (n=n-2 x=y) */
588  y = vec_sr(x, s);
589  nt = vec_sub(n,s);
590  m = (vui16_t)vec_cmpgt(y, z);
591  s = vec_sr(s,one);
592  x = vec_sel (x , y, m);
593  n = vec_sel (n , nt, m);
594 
595  /* y=x>>1 if (y!=0) return (n=n-2) */
596  y = vec_sr(x, s);
597  nt = vec_sub(n,s);
598  nt = vec_sub(nt,s);
599  m = (vui16_t)vec_cmpgt(y, z);
600  n = vec_sel (n , nt, m);
601 
602  /* else return (x-n) */
603  nt = vec_sub (n, x);
604  r = vec_sel (nt , n, m);
605 #endif
606 
607  return (r);
608 }

825 static inline vui16_t
826 vec_popcnth (vui16_t vra)
827 {
828  vui16_t r;
829 #ifdef _ARCH_PWR8
830 #ifndef vec_vpopcnth
831  __asm__(
832  "vpopcnth %0,%1;"
833  : "=v" (r)
834  : "v" (vra)
835  : );
836 #else
837  r = vec_vpopcnth (vra);
838 #endif
839 #else
840  //#warning Implememention pre power8
841  __vector unsigned short n, x1, x2, x, s;
842  __vector unsigned short ones = { 1,1,1,1, 1,1,1,1};
843  __vector unsigned short fives =
844  {0x5555,0x5555,0x5555,0x5555, 0x5555,0x5555,0x5555,0x5555};
845  __vector unsigned short threes =
846  {0x3333,0x3333,0x3333,0x3333, 0x3333,0x3333,0x3333,0x3333};
847  __vector unsigned short fs =
848  {0x0f0f,0x0f0f,0x0f0f,0x0f0f, 0x0f0f,0x0f0f,0x0f0f,0x0f0f};
849  /* n = 8 s = 4 */
850  s = ones;
851  x = vra;
852 
853  /* x = x - ((x >> 1) & 0x5555) */
854  x2 = vec_and (vec_sr (x, s), fives);
855  n = vec_sub (x, x2);
856  s = vec_add (s, s);
857 
858  /* x = (x & 0x3333) + ((x & 0xcccc) >> 2) */
859  x1 = vec_and (n, threes);
860  x2 = vec_andc (n, threes);
861  n = vec_add (x1, vec_sr (x2, s));
862  s = vec_add (s, s);
863 
864  /* x = (x + (x >> 4)) & 0x0f0f) */
865  x1 = vec_add (n, vec_sr (n, s));
866  n = vec_and (x1, fs);
867  s = vec_add (s, s);
868 
869  /* This avoids the extra load const. */
870  /* x = (x + (x << 8)) >> 8) */
871  x1 = vec_add (n, vec_sl (n, s));
872  r = vec_sr (x1, s);
873 #endif
874  return (r);
875 }

941 static inline vui16_t
942 vec_slhi (vui16_t vra, const unsigned int shb)
943 {
944  vui16_t lshift;
945  vui16_t result;
946 
947  if (shb < 16)
948  {
949  /* Load the shift const in a vector. The element shifts require
950  a shift amount for each element. For the immediate form the
951  shift constant is splatted to all elements of the
952  shift control. */
953  if (__builtin_constant_p(shb))
954  lshift = (vui16_t) vec_splat_s16(shb);
955  else
956  lshift = vec_splats ((unsigned short) shb);
957 
958  /* Vector Shift right bytes based on the lower 4-bits of
959  corresponding element of lshift. */
960  result = vec_vslh (vra, lshift);
961  }
962  else
963  { /* shifts greater then 15 bits return zeros. */
964  result = vec_xor ((vui16_t) vra, (vui16_t) vra);
965  }
966 
967  return (vui16_t) result;
968 }
