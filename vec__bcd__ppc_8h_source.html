261 static inline vui8_t
262 vec_bcdctb100s (vui8_t vra)
263 {
264  vui8_t result;
265  vui8_t high_digit;
266  vui8_t j, k, l, m;
267  const vui8_t e_perm =
268 #if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
269  { 0x00, 0x10, 0x02, 0x12, 0x04, 0x14, 0x06, 0x16, 0x08, 0x18, 0x0a, 0x1a,
270  0x0c, 0x1c, 0x0e, 0x1e };
271 #else
272  { 0x01, 0x11, 0x03, 0x13, 0x05, 0x15, 0x07, 0x17, 0x09, 0x19, 0x0b, 0x1b,
273  0x0d, 0x1d, 0x0f, 0x1f };
274 #endif
275 
276  high_digit = vec_srbi (vra, 4);
277 
278  /* Compute the high digit correction factor. For BCD to binary 100s
279  * this the isolated high digit multiplied by the radix different in
280  * binary. For this stage we use 0x10 - 10 = 6. */
281  k = vec_splat_u8((unsigned char )0x06);
282  l = (vui8_t) vec_mule (high_digit, k);
283  m = (vui8_t) vec_mulo (high_digit, k);
284  j = vec_perm (l, m, e_perm);
285 
286  /* Subtract the high digit correction bytes from the original
287  * BCD bytes values in binary. */
288  result = vec_sub (vra, j);
289 
290  return result;
291 }
