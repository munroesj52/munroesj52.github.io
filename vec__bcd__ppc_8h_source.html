1924 vec_bcdcfz (vui8_t vrb)
1925 {
1926  vBCD_t vrt;
1927 #ifdef _ARCH_PWR9
1928  __asm__(
1929  "bcdcfz. %0,%1,0;\n"

1931  : "v" (vrb)
1932  : "cr6" );
1933 #else
1934  const vui8_t dmask = vec_splat_u8(15);
1935  const vui8_t dx0 = vec_splat_u8(0);
1936  vui8_t znd_s;
1937  vui8_t znd_d, znd_t;
1938  vui8_t bcd, bcd_h, bcd_l;
1939  // Isolate the BCD digit from each zoned character.
1940  znd_d = vec_and (vrb, dmask);
1941  znd_t = (vui8_t) vec_srqi ((vui128_t) znd_d, 4);
1942  // Isolate the bit (1) that matters in the Zoned sign code.
1943  znd_s = vec_slbi (vrb, 1);
1944  znd_s = vec_srbi (znd_s, 7);
1945  // Convert to BCD preferred sign code 0xC or 0xD
1946  znd_s = vec_or (znd_s, (vui8_t) _BCD_CONST_ZERO);
1947  // Pack the even/odd zone digits into a single vector.
1948 #if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
1949  bcd = vec_pack ((vui16_t) znd_d, (vui16_t) znd_t);
1950 #else
1951  bcd = vec_pack ((vui16_t) znd_t, (vui16_t) znd_d);
1952 #endif
1953  // Swap even/odd DWs to low half and OR to get unsigned 16-digit BCD.
1954  bcd_l = (vui8_t) vec_mrgald ((vui128_t) dx0, (vui128_t) bcd);
1955  bcd_h = (vui8_t) vec_mrgahd ((vui128_t) dx0, (vui128_t) bcd);

1959  // Merge sign code from zone code.
1960  vrt = vec_bcdcpsgn (vrt, (vBCD_t) znd_s);
