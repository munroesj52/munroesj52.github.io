191 static inline
192 vui64_t
193 vec_cmpequd (vui64_t a, vui64_t b)
194 {
195  vui64_t result;
196 #ifdef _ARCH_PWR8
197 #if __GNUC__ >= 7
198  result = (vui64_t)vec_cmpeq((__vector bool long long)a, (__vector bool long long)b);
199 #else
200  __asm__(
201  "vcmpequd %0,%1,%2;\n"
202  : "=v" (result)
203  : "v" (a),
204  "v" (b)
205  : );
206 #endif
207 #else
208  /*
209  * Don't have vector compare equal unsigned doubleword until power8.
210  * So we have to compare word and unless all_eq we need to do some
211  * extra work, ie the words may have different truth values. So we
212  * rotate each doubleword by 32-bits (here we use permute as we don't
213  * have rotate doubleword either). Then vand the original word
214  * compare and rotated value to get the final value.
215  */
216  vui8_t permute =
217  { 0x04,0x05,0x6,0x7, 0x00,0x01,0x2,0x03, 0x0C,0x0D,0x0E,0x0F, 0x08,0x09,0x0A,0x0B};
218  vui32_t r, rr;
219  r = (vui32_t)vec_cmpeq((vui32_t)a, (vui32_t)b);
220  if (vec_any_ne((vui32_t)a, (vui32_t)b))
221  {
222  rr = vec_perm (r, r, permute);
223  r= vec_and (r, rr);
224  }
225  result = (vui64_t)r;
226 #endif
227  return (result);
228 }

278 static inline
279 vui64_t
280 vec_cmpgtud (vui64_t a, vui64_t b)
281 {
282  vui64_t result;
283 #ifdef _ARCH_PWR8
284 #if __GNUC__ >= 7
285  result = (vui64_t)vec_cmpgt(a, b);
286 #else
287  __asm__(
288  "vcmpgtud %0,%1,%2;\n"
289  : "=v" (result)
290  : "v" (a),
291  "v" (b)
292  : );
293 #endif
294 #else
295  /*
296  * Don't have vector compare greater than unsigned doubleword until
297  * power8. So we have to use compare word and logic to compute the
298  * doubleword truth values.
299  */
300  __vector unsigned int r, x, y;
301  __vector unsigned int c0, c1, c01;
302  __vector unsigned int eq, gt, a32, b32;
303 
304  /* c10 = {0, -1, 0, -1} */
305  c0 = vec_splat_u32 (0);
306  c1 = vec_splat_u32 (-1);
307  c01 = vec_mergeh (c0, c1);
308 
309  a32 = (__vector unsigned int)a;
310  b32 = (__vector unsigned int)b;
311 
312  gt = (__vector unsigned int)vec_cmpgt (a32, b32);
313  eq = (__vector unsigned int)vec_cmpeq (a32, b32);
314  /* GTxw = GThw | (EQhw & GTlw) */
315  x = vec_sld (gt, c0, 4);
316  y = vec_and (eq, x);
317  x = vec_or (gt, y);
318  /* Duplicate result word to dword width. */
319  y = vec_sld (c0, x, 12);
320  r = vec_sel (x, y, c01);
321  result = (vui64_t)r;
322 #endif
323  return (result);
324 }

572 static inline vui64_t
573 vec_permdi (vui64_t vra, vui64_t vrb, const int ctl)
574 {
575  vui64_t result;
576 #ifdef _ARCH_PWR7
577  switch (ctl & 3)
578  {
579 #if __BYTE_ORDER__ == __ORDER_BIG_ENDIAN__
580  case 0:
581  result = vec_xxpermdi (vra, vrb, 0);
582  break;
583  case 1:
584  result = vec_xxpermdi (vra, vrb, 1);
585  break;
586  case 2:
587  result = vec_xxpermdi (vra, vrb, 2);
588  break;
589  case 3:
590  result = vec_xxpermdi (vra, vrb, 3);
591  break;
592 #else
593  case 0:
594  result = vec_xxpermdi (vrb, vra, 3);
595  break;
596  case 1:
597  result = vec_xxpermdi (vrb, vra, 1);
598  break;
599  case 2:
600  result = vec_xxpermdi (vrb, vra, 2);
601  break;
602  case 3:
603  result = vec_xxpermdi (vrb, vra, 0);
604  break;
605 #endif
606  default:
607  result = (vui64_t){ 0, 0 };
608  }
609 #else
610  /* Current compilers don't accept vector unsigned long int as vector
611  * parms to vec_sld, so use vector unsigned int. The vsldoi
612  * instruction does not care). */
613  vui32_t temp;
614  switch (ctl & 3)
615  {
616  case 0:
617  temp = vec_sld ((vui32_t) vra, (vui32_t) vra, 8);
618  result = (vui64_t) vec_sld (temp, (vui32_t) vrb, 8);
619  break;
620  case 1:
621  temp = vec_sld ((vui32_t) vrb, (vui32_t) vra, 8);
622  result = (vui64_t) vec_sld (temp, temp, 8);
623  break;
624  case 2:
625  result = (vui64_t) vec_sld ((vui32_t) vra, (vui32_t) vrb, 8);
626  break;
627  case 3:
628  temp = vec_sld ((vui32_t) vrb, (vui32_t) vrb, 8);
629  result = (vui64_t) vec_sld ((vui32_t) vra, temp, 8);
630  break;
631  }
632 #endif
633  return (result);

737 static inline vui64_t
738 vec_sldi (vui64_t vra, const unsigned int shb)
739 {
740  vui64_t lshift;
741  vui64_t result;
742 
743  if (shb < 64)
744  {
745  /* Load the shift const in a vector. The element shifts require
746  a shift amount for each element. For the immediate form the
747  shift constant is splatted to all elements of the
748  shift control. */
749  if (__builtin_constant_p (shb) && (shb < 16))
750  lshift = (vui64_t) vec_splat_s32(shb);
751  else
752  lshift = vec_splats ((unsigned long) shb);
753 
754  /* Vector Shift right bytes based on the lower 6-bits of
755  corresponding element of lshift. */
756  result = vec_vsld (vra, lshift);
757  }
758  else
759  { /* shifts greater then 31 bits return zeros. */
760  result = vec_xor ((vui64_t) vra, (vui64_t) vra);
761  }
762 
763  return (vui64_t) result;
764 }

968 #ifndef vec_vsld
969 static inline vui64_t
970 vec_vsld (vui64_t vra, vui64_t vrb)
971 {
972  vui64_t result;
973 
974 #ifdef _ARCH_PWR8
975  __asm__(
976  "vsld %0,%1,%2;"
977  : "=v" (result)
978  : "v" (vra),
979  "v" (vrb)
980  : );
981 #else
982  vui8_t vsh_h, vsh_l;
983  vui8_t vr_h, vr_l;
984  vui64_t sel_mask = CONST_VINT128_DW(0, -1LL);
985  vui64_t shft_mask = CONST_VINT128_DW(63, 63);
986 
987  /* constrain the dword shift amounts to 0-63. */
988  vsh_l = vec_and ((vui8_t)vrb, (vui8_t)shft_mask);
989  /* Isolate the high dword so that bits from the low dword
990  * do not contaminate the result. */
991  vr_h = vec_andc ((vui8_t)vra, (vui8_t)sel_mask);
992  /* The low dword is just vra as the 128-bit shift left generates
993  * '0's on the right and the final merge (vec_sel)
994  * cleans up 64-bit overflow on the left. */
995  vr_l = (vui8_t)vra;
996  /* The vsr instruction only works correctly if the bit shift
997  * value is splatted to each byte of the vector. */
998  vsh_h = vec_splat (vsh_l, VEC_BYTE_L_DWH);
999  vsh_l = vec_splat (vsh_l, VEC_BYTE_L_DWL);
1000 // vsht_splat = vec_splat ((vui8_t) vrb, VEC_BYTE_L);
1001  /* Shift the high dword by vsh_h. */
1002  vr_h = vec_vslo (vr_h, vsh_h);
1003  vr_h = vec_vsl (vr_h, vsh_h);
1004  /* Shift the low dword by vsh_l. */
1005  vr_l = vec_vslo (vr_l, vsh_l);
