1111 static inline
1112 vb64_t
1113 vec_cmpequd (vui64_t a, vui64_t b)
1114 {
1115  vb64_t result;
1116 #ifdef _ARCH_PWR8
1117 #if __GNUC__ >= 6
1118  result = vec_cmpeq(a, b);
1119 #else
1120  __asm__(
1121  "vcmpequd %0,%1,%2;\n"
1122  : "=v" (result)
1123  : "v" (a),
1124  "v" (b)
1125  : );
1126 #endif
1127 #else
1128  /*
1129  * Don't have vector compare equal unsigned doubleword until power8.
1130  * So we have to compare word and unless all_eq we need to do some
1131  * extra work, ie the words may have different truth values. So we
1132  * rotate each doubleword by 32-bits (here we use permute as we don't
1133  * have rotate doubleword either). Then vand the original word
1134  * compare and rotated value to get the final value.
1135  */
1136  vui8_t permute =
1137  { 0x04,0x05,0x6,0x7, 0x00,0x01,0x2,0x03, 0x0C,0x0D,0x0E,0x0F, 0x08,0x09,0x0A,0x0B};
1138  vui32_t r, rr;
1139  r = (vui32_t) vec_cmpeq ((vui32_t) a, (vui32_t) b);
1140  if (vec_any_ne ((vui32_t) a, (vui32_t) b))
1141  {
1142  rr = vec_perm (r, r, permute);
1143  r= vec_and (r, rr);
1144  }
1145  result = (vb64_t)r;
1146 #endif
1147  return (result);
1148 }

1282 static inline
1283 vb64_t
1284 vec_cmpgtud (vui64_t a, vui64_t b)
1285 {
1286  vb64_t result;
1287 #ifdef _ARCH_PWR8
1288 #if __GNUC__ >= 6
1289  result = vec_cmpgt(a, b);
1290 #else
1291  __asm__(
1292  "vcmpgtud %0,%1,%2;\n"
1293  : "=v" (result)
1294  : "v" (a),
1295  "v" (b)
1296  : );
1297 #endif
1298 #else
1299  /*
1300  * Don't have vector compare greater than unsigned doubleword until
1301  * power8. So we have to use compare word and logic to compute the
1302  * doubleword truth values.
1303  */
1304  __vector unsigned int r, x, y;
1305  __vector unsigned int c0, c1, c01;
1306  __vector unsigned int eq, gt, a32, b32;
1307 
1308  /* c10 = {0, -1, 0, -1} */
1309  c0 = vec_splat_u32 (0);
1310  c1 = vec_splat_u32 (-1);
1311  c01 = vec_mergeh (c0, c1);
1312 
1313  a32 = (__vector unsigned int)a;
1314  b32 = (__vector unsigned int)b;
1315 
1316  gt = (__vector unsigned int)vec_cmpgt (a32, b32);
1317  eq = (__vector unsigned int)vec_cmpeq (a32, b32);
1318  /* GTxw = GThw | (EQhw & GTlw) */
1319  x = vec_sld (gt, c0, 4);
1320  y = vec_and (eq, x);
1321  x = vec_or (gt, y);
1322  /* Duplicate result word to dword width. */
1323  y = vec_sld (c0, x, 12);
1324  r = vec_sel (x, y, c01);
1325  result = (vb64_t)r;
1326 #endif
1327  return (result);
1328 }
