125 vui64_t
126 vec_addudm(vui64_t a, vui64_t b)
127 {
128  vui32_t r;
129 
130 #ifdef _ARCH_PWR8
131 #ifndef vec_vaddudm
132  __asm__(
133  "vaddudm %0,%1,%2;"
134  : "=v" (r)
135  : "v" (a),
136  "v" (b)
137  : );
138 #else
139  r = (vui32_t) vec_vaddudm (a, b);
140 #endif
141 #else
142  vui32_t c;
143  vui32_t z= { 0,0,0,0};
144  vui32_t cm= { 0,1,0,1};
145 
146  c = vec_vaddcuw ((vui32_t)a, (vui32_t)b);
147  r = vec_vadduwm ((vui32_t)a, (vui32_t)b);
148  c = vec_and (c, cm);
149  c = vec_sld (c, z, 4);
150  r = vec_vadduwm (r, c);
151 #endif
152  return ((vui64_t) r);
153 }

260 static inline
261 vb64_t
262 vec_cmpequd (vui64_t a, vui64_t b)
263 {
264  vb64_t result;
265 #ifdef _ARCH_PWR8
266 #if __GNUC__ >= 6
267  result = vec_cmpeq(a, b);
268 #else
269  __asm__(
270  "vcmpequd %0,%1,%2;\n"
271  : "=v" (result)
272  : "v" (a),
273  "v" (b)
274  : );
275 #endif
276 #else
277  /*
278  * Don't have vector compare equal unsigned doubleword until power8.
279  * So we have to compare word and unless all_eq we need to do some
280  * extra work, ie the words may have different truth values. So we
281  * rotate each doubleword by 32-bits (here we use permute as we don't
282  * have rotate doubleword either). Then vand the original word
283  * compare and rotated value to get the final value.
284  */
285  vui8_t permute =
286  { 0x04,0x05,0x6,0x7, 0x00,0x01,0x2,0x03, 0x0C,0x0D,0x0E,0x0F, 0x08,0x09,0x0A,0x0B};
287  vui32_t r, rr;
288  r = (vui32_t)vec_cmpeq((vui32_t)a, (vui32_t)b);
289  if (vec_any_ne((vui32_t)a, (vui32_t)b))
290  {
291  rr = vec_perm (r, r, permute);
292  r= vec_and (r, rr);
293  }
294  result = (vb64_t)r;
295 #endif
296  return (result);
297 }

431 static inline
432 vb64_t
433 vec_cmpgtud (vui64_t a, vui64_t b)
434 {
435  vb64_t result;
436 #ifdef _ARCH_PWR8
437 #if __GNUC__ >= 6
438  result = vec_cmpgt(a, b);
439 #else
440  __asm__(
441  "vcmpgtud %0,%1,%2;\n"
442  : "=v" (result)
443  : "v" (a),
444  "v" (b)
445  : );
446 #endif
447 #else
448  /*
449  * Don't have vector compare greater than unsigned doubleword until
450  * power8. So we have to use compare word and logic to compute the
451  * doubleword truth values.
452  */
453  __vector unsigned int r, x, y;
454  __vector unsigned int c0, c1, c01;
455  __vector unsigned int eq, gt, a32, b32;
456 
457  /* c10 = {0, -1, 0, -1} */
458  c0 = vec_splat_u32 (0);
459  c1 = vec_splat_u32 (-1);
460  c01 = vec_mergeh (c0, c1);
461 
462  a32 = (__vector unsigned int)a;
463  b32 = (__vector unsigned int)b;
464 
465  gt = (__vector unsigned int)vec_cmpgt (a32, b32);
466  eq = (__vector unsigned int)vec_cmpeq (a32, b32);
467  /* GTxw = GThw | (EQhw & GTlw) */
468  x = vec_sld (gt, c0, 4);
469  y = vec_and (eq, x);
470  x = vec_or (gt, y);
471  /* Duplicate result word to dword width. */
472  y = vec_sld (c0, x, 12);
473  r = vec_sel (x, y, c01);
474  result = (vb64_t)r;
475 #endif
476  return (result);
477 }
