290 static inline
291 vb64_t
292 vec_cmpequd (vui64_t a, vui64_t b)
293 {
294  vb64_t result;
295 #ifdef _ARCH_PWR8
296 #if __GNUC__ >= 6
297  result = vec_cmpeq(a, b);
298 #else
299  __asm__(
300  "vcmpequd %0,%1,%2;\n"
301  : "=v" (result)
302  : "v" (a),
303  "v" (b)
304  : );
305 #endif
306 #else
307  /*
308  * Don't have vector compare equal unsigned doubleword until power8.
309  * So we have to compare word and unless all_eq we need to do some
310  * extra work, ie the words may have different truth values. So we
311  * rotate each doubleword by 32-bits (here we use permute as we don't
312  * have rotate doubleword either). Then vand the original word
313  * compare and rotated value to get the final value.
314  */
315  vui8_t permute =
316  { 0x04,0x05,0x6,0x7, 0x00,0x01,0x2,0x03, 0x0C,0x0D,0x0E,0x0F, 0x08,0x09,0x0A,0x0B};
317  vui32_t r, rr;
318  r = (vui32_t)vec_cmpeq((vui32_t)a, (vui32_t)b);
319  if (vec_any_ne((vui32_t)a, (vui32_t)b))
320  {
321  rr = vec_perm (r, r, permute);
322  r= vec_and (r, rr);
323  }
324  result = (vb64_t)r;
325 #endif
326  return (result);
327 }

461 static inline
462 vb64_t
463 vec_cmpgtud (vui64_t a, vui64_t b)
464 {
465  vb64_t result;
466 #ifdef _ARCH_PWR8
467 #if __GNUC__ >= 6
468  result = vec_cmpgt(a, b);
469 #else
470  __asm__(
471  "vcmpgtud %0,%1,%2;\n"
472  : "=v" (result)
473  : "v" (a),
474  "v" (b)
475  : );
476 #endif
477 #else
478  /*
479  * Don't have vector compare greater than unsigned doubleword until
480  * power8. So we have to use compare word and logic to compute the
481  * doubleword truth values.
482  */
483  __vector unsigned int r, x, y;
484  __vector unsigned int c0, c1, c01;
485  __vector unsigned int eq, gt, a32, b32;
486 
487  /* c10 = {0, -1, 0, -1} */
488  c0 = vec_splat_u32 (0);
489  c1 = vec_splat_u32 (-1);
490  c01 = vec_mergeh (c0, c1);
491 
492  a32 = (__vector unsigned int)a;
493  b32 = (__vector unsigned int)b;
494 
495  gt = (__vector unsigned int)vec_cmpgt (a32, b32);
496  eq = (__vector unsigned int)vec_cmpeq (a32, b32);
497  /* GTxw = GThw | (EQhw & GTlw) */
498  x = vec_sld (gt, c0, 4);
499  y = vec_and (eq, x);
500  x = vec_or (gt, y);
501  /* Duplicate result word to dword width. */
502  y = vec_sld (c0, x, 12);
503  r = vec_sel (x, y, c01);
504  result = (vb64_t)r;
505 #endif
506  return (result);
507 }
