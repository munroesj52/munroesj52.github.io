1471 static inline vf64_t
1472 vec_vlxsfdx (const signed long long ra, const double *rb)
1473 {
1474  vf64_t xt;
1475 
1476 #if defined (__clang__)
1477  __VEC_U_128 t;
1478  unsigned long long *p = (unsigned long long *)((char *)rb + ra);
1479  t.ulong.upper = *p;
1480  xt = t.vx1;
1481 #else
1482  if (__builtin_constant_p (ra) && (ra < 32760) && (ra >= -32768)
1483  && ((ra & 3) == 0))
1484  {
1485 #if defined (_ARCH_PWR9)
1486  __asm__(
1487  "lxsd%X1 %0,%1;"
1488  : "=v" (xt)
1489  : "m" (*(double*)((char *)rb + ra))
1490  : );
1491 #else
1492  if (ra == 0)
1493  {
1494  __asm__(
1495  "lxsdx %x0,%y1;"
1496  : "=wa" (xt)
1497  : "Z" (*rb)
1498  : );
1499  } else {
1500  unsigned long long rt;
1501 #if defined (_ARCH_PWR8)
1502  // For P8 better if li and lxsdx shared a single asm block
1503  // (enforcing consecutive instructions).
1504  // This enables instruction fusion for P8.
1505  __asm__(
1506  "li %0,%2;"
1507  "lxsdx %x1,%3,%0;"
1508  : "=&r" (rt), "=wa" (xt)
1509  : "I" (ra), "b" (rb), "m" (*(double*)((char *)rb+ra))
1510  : );
1511 #else // _ARCH_PWR7
1512  // This generates operationally the same code, but the
1513  // compiler may rearrange/schedule the code.
1514  __asm__(
1515  "li %0,%1;"
1516  : "=r" (rt)
1517  : "I" (ra)
1518  : );
1519  __asm__(
1520  "lxsdx %x0,%y1;"
1521  : "=wa" (xt)
1522  : "Z" (*(double*)((char *)rb+rt))
1523  : );
1524 #endif
1525  }
1526 #endif
1527  } else {
1528  __asm__(
1529  "lxsdx %x0,%y1;"
1530  : "=wa" (xt)
1531  : "Z" (*(double*)((char *)rb+ra))
1532  : );
1533  }
1534 #endif
1535  return xt;
