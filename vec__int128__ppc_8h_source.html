500 static inline vui128_t
501 vec_clzq (vui128_t vra)
502 {
503  vui64_t result;
504 
505 #ifdef _ARCH_PWR8
506  /*
507  * Use the Vector Count Leading Zeros Double Word instruction to get
508  * the count for the left and right vector halves. If the left vector
509  * doubleword of the input is nonzero then only the left count is
510  * included and we need to mask off the right count.
511  * Otherwise the left count is 64 and we need to add 64 to the right
512  * count.
513  * After masking we sum across the left and right counts to
514  * get the final 128-bit vector count (0-128).
515  */
516  vui64_t vt1, vt2, vt3;
517  vui64_t vzero = { 0, 0 };
518  vui64_t v64 = { 64, 64 };
519 
520  vt1 = vec_vclz ((vui64_t) vra);
521  vt2 = (vui64_t) vec_cmplt(vt1, v64);
522  vt3 = (vui64_t) vec_sld ((vui8_t) vzero, (vui8_t) vt2, 8);
523  result = vec_andc (vt1, vt3);
524  result = (vui64_t) vec_sums ((vi32_t) result, (vi32_t) vzero);
525 #if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
526  result = (vui64_t) vec_sld ((vui8_t) result, (vui8_t) result, 4);
527 #endif
528 #else
529  /* vector clz instructions were introduced in power8. For power7 and
530  * earlier, use the pveclib vec_clzw implementation. For a quadword
531  * clz, this requires pre-conditioning the input before computing the
532  * the word clz and sum across. */
533  vui32_t c0, clz;
534  vui32_t r32, gt32, gt32sr32, gt64sr64;
535 
536  c0 = vec_splat_u32 (0);
537  gt32 = (vui32_t)vec_cmpgt((vui32_t)vra, c0);
538  gt32sr32 = vec_sld (c0, gt32, 12);
539  gt64sr64 = vec_sld (c0, gt32, 8);
540  gt32 = vec_sld (c0, gt32, 4);
541 
542  gt32sr32 = vec_or (gt32sr32, gt32);
543  gt64sr64 = vec_or (gt64sr64, (vui32_t)vra);
544  r32 = vec_or (gt32sr32, gt64sr64);
545 
546  clz = vec_clzw (r32);
547  result = (vui64_t)vec_sums ((vi32_t)clz, (vi32_t)c0);
548 #endif
549 
550  return ((vui128_t) result);
551 }
