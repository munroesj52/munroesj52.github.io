941 static inline vui128_t
942 vec_clzq (vui128_t vra)
943 {
944  vui64_t result;
945 
946 #ifdef _ARCH_PWR8
947  /*
948  * Use the Vector Count Leading Zeros Double Word instruction to get
949  * the count for the left and right vector halves. If the left vector
950  * doubleword of the input is nonzero then only the left count is
951  * included and we need to mask off the right count.
952  * Otherwise the left count is 64 and we need to add 64 to the right
953  * count.
954  * After masking we sum across the left and right counts to
955  * get the final 128-bit vector count (0-128).
956  */
957  vui64_t vt1, vt2, vt3;
958  vui64_t vzero = { 0, 0 };
959  vui64_t v64 = { 64, 64 };
960 
961  vt1 = vec_vclz ((vui64_t) vra);
962  vt2 = (vui64_t) vec_cmplt(vt1, v64);
963  vt3 = (vui64_t) vec_sld ((vui8_t) vzero, (vui8_t) vt2, 8);
964  result = vec_andc (vt1, vt3);
965  result = (vui64_t) vec_sums ((vi32_t) result, (vi32_t) vzero);
966 #if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
967  result = (vui64_t) vec_sld ((vui8_t) result, (vui8_t) result, 4);
968 #endif
969 #else
970  /* vector clz instructions were introduced in power8. For power7 and
971  * earlier, use the pveclib vec_clzw implementation. For a quadword
972  * clz, this requires pre-conditioning the input before computing the
973  * the word clz and sum across. */
974  vui32_t c0, clz;
975  vui32_t r32, gt32, gt32sr32, gt64sr64;
976 
977  c0 = vec_splat_u32 (0);
978  gt32 = (vui32_t) vec_cmpgt ((vui32_t) vra, c0);
979  gt32sr32 = vec_sld (c0, gt32, 12);
980  gt64sr64 = vec_sld (c0, gt32, 8);
981  gt32 = vec_sld (c0, gt32, 4);
982 
983  gt32sr32 = vec_or (gt32sr32, gt32);
984  gt64sr64 = vec_or (gt64sr64, (vui32_t) vra);
985  r32 = vec_or (gt32sr32, gt64sr64);
986 
987  clz = vec_clzw (r32);
988  result = (vui64_t) vec_sums ((vi32_t) clz, (vi32_t) c0);
989 #endif
990 
991  return ((vui128_t) result);
992 }
