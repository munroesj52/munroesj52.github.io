1347 static inline vui128_t
1348 vec_clzq (vui128_t vra)
1349 {
1350  vui64_t result;
1351 
1352 #ifdef _ARCH_PWR8
1353  /*
1354  * Use the Vector Count Leading Zeros Double Word instruction to get
1355  * the count for the left and right vector halves. If the left vector
1356  * doubleword of the input is nonzero then only the left count is
1357  * included and we need to mask off the right count.
1358  * Otherwise the left count is 64 and we need to add 64 to the right
1359  * count.
1360  * After masking we sum across the left and right counts to
1361  * get the final 128-bit vector count (0-128).
1362  */
1363  vui64_t vt1, vt2, vt3;
1364  vui64_t vzero = { 0, 0 };
1365  vui64_t v64 = { 64, 64 };
1366 
1367  vt1 = vec_vclz ((vui64_t) vra);
1368  vt2 = (vui64_t) vec_cmplt(vt1, v64);
1369  vt3 = (vui64_t) vec_sld ((vui8_t) vzero, (vui8_t) vt2, 8);
1370  result = vec_andc (vt1, vt3);
1371  result = (vui64_t) vec_sums ((vi32_t) result, (vi32_t) vzero);
1372 #if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
1373  result = (vui64_t) vec_sld ((vui8_t) result, (vui8_t) result, 4);
1374 #endif
1375 #else
1376  /* vector clz instructions were introduced in power8. For power7 and
1377  * earlier, use the pveclib vec_clzw implementation. For a quadword
1378  * clz, this requires pre-conditioning the input before computing the
1379  * the word clz and sum across. */
1380  vui32_t c0, clz;
1381  vui32_t r32, gt32, gt32sr32, gt64sr64;
1382 
1383  c0 = vec_splat_u32 (0);
1384  gt32 = (vui32_t) vec_cmpgt ((vui32_t) vra, c0);
1385  gt32sr32 = vec_sld (c0, gt32, 12);
1386  gt64sr64 = vec_sld (c0, gt32, 8);
1387  gt32 = vec_sld (c0, gt32, 4);
1388 
1389  gt32sr32 = vec_or (gt32sr32, gt32);
1390  gt64sr64 = vec_or (gt64sr64, (vui32_t) vra);
1391  r32 = vec_or (gt32sr32, gt64sr64);
1392 
1393  clz = vec_clzw (r32);
1394  result = (vui64_t) vec_sums ((vi32_t) clz, (vi32_t) c0);
1395 #endif
1396 
1397  return ((vui128_t) result);
1398 }
