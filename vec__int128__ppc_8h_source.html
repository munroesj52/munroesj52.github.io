2487 static inline vui128_t
2488 vec_clzq (vui128_t vra)
2489 {

2491 
2492 #ifdef _ARCH_PWR8
2493  /*
2494  * Use the Vector Count Leading Zeros Double Word instruction to get
2495  * the count for the left and right vector halves. If the left vector
2496  * doubleword of the input is nonzero then only the left count is
2497  * included and we need to mask off the right count.
2498  * Otherwise the left count is 64 and we need to add 64 to the right
2499  * count.
2500  * After masking we sum across the left and right counts to
2501  * get the final 128-bit vector count (0-128).
2502  */
2503  vui64_t vt1, vt2, vt3, h64, l64;
2504  const vui64_t vzero = { 0, 0 };
2505 
2506  vt1 = vec_clzd ((vui64_t) vra);
2507  vt2 = (vui64_t) vec_cmpequd((vui64_t) vra, vzero);
2508  vt3 = vec_mrgahd ((vui128_t)vzero, (vui128_t)vt2);
2509  h64 = vec_mrgahd ((vui128_t)vzero, (vui128_t)vt1);
2510  l64 = vec_and (vt1, vt3);
2511  result = vec_addudm (h64, l64);
2512 #else
2513  /* vector clz instructions were introduced in power8. For power7 and
2514  * earlier, use the pveclib vec_clzw implementation. For a quadword
2515  * clz, this requires pre-conditioning the input before computing the
2516  * the word clz and sum across. */
2517  vui32_t c0, clz;
2518  vui32_t r32, gt32, gt32sr32, gt64sr64;
2519

2521  gt32 = (vui32_t) vec_cmpgt ((vui32_t) vra, c0);
2522  gt32sr32 = vec_sld (c0, gt32, 12);
2523  gt64sr64 = vec_sld (c0, gt32, 8);
2524  gt32 = vec_sld (c0, gt32, 4);
2525 
2526  gt32sr32 = vec_or (gt32sr32, gt32);
2527  gt64sr64 = vec_or (gt64sr64, (vui32_t) vra);
2528  r32 = vec_or (gt32sr32, gt64sr64);
2529 
2530  clz = vec_clzw (r32);
2531  result = (vui64_t) vec_sums ((vi32_t) clz, (vi32_t) c0);
2532 #endif
2533
