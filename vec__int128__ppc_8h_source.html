2511 static inline vui128_t
2512 vec_clzq (vui128_t vra)
2513 {
2514  vui64_t result;
2515 
2516 #ifdef _ARCH_PWR8
2517  /*
2518  * Use the Vector Count Leading Zeros Double Word instruction to get
2519  * the count for the left and right vector halves. If the left vector
2520  * doubleword of the input is nonzero then only the left count is
2521  * included and we need to mask off the right count.
2522  * Otherwise the left count is 64 and we need to add 64 to the right
2523  * count.
2524  * After masking we sum across the left and right counts to
2525  * get the final 128-bit vector count (0-128).
2526  */
2527  vui64_t vt1, vt2, vt3, h64, l64;
2528  const vui64_t vzero = { 0, 0 };
2529 
2530  vt1 = vec_clzd ((vui64_t) vra);
2531  vt2 = (vui64_t) vec_cmpequd((vui64_t) vra, vzero);
2532  vt3 = vec_mrgahd ((vui128_t)vzero, (vui128_t)vt2);
2533  h64 = vec_mrgahd ((vui128_t)vzero, (vui128_t)vt1);
2534  l64 = vec_and (vt1, vt3);
2535  result = vec_addudm (h64, l64);
2536 #else
2537  /* vector clz instructions were introduced in power8. For power7 and
2538  * earlier, use the pveclib vec_clzw implementation. For a quadword
2539  * clz, this requires pre-conditioning the input before computing the
2540  * the word clz and sum across. */
2541  vui32_t c0, clz;
2542  vui32_t r32, gt32, gt32sr32, gt64sr64;
2543 
2544  c0 = vec_splat_u32 (0);
2545  gt32 = (vui32_t) vec_cmpgt ((vui32_t) vra, c0);
2546  gt32sr32 = vec_sld (c0, gt32, 12);
2547  gt64sr64 = vec_sld (c0, gt32, 8);
2548  gt32 = vec_sld (c0, gt32, 4);
2549 
2550  gt32sr32 = vec_or (gt32sr32, gt32);
2551  gt64sr64 = vec_or (gt64sr64, (vui32_t) vra);
2552  r32 = vec_or (gt32sr32, gt64sr64);
2553 
2554  clz = vec_clzw (r32);
2555  result = (vui64_t) vec_sums ((vi32_t) clz, (vi32_t) c0);
2556 #endif
2557 
2558  return ((vui128_t) result);
2559 }
