1613 static inline vui128_t
1614 vec_clzq (vui128_t vra)
1615 {
1616  vui64_t result;
1617 
1618 #ifdef _ARCH_PWR8
1619  /*
1620  * Use the Vector Count Leading Zeros Double Word instruction to get
1621  * the count for the left and right vector halves. If the left vector
1622  * doubleword of the input is nonzero then only the left count is
1623  * included and we need to mask off the right count.
1624  * Otherwise the left count is 64 and we need to add 64 to the right
1625  * count.
1626  * After masking we sum across the left and right counts to
1627  * get the final 128-bit vector count (0-128).
1628  */
1629  vui64_t vt1, vt2, vt3;
1630  vui64_t vzero = { 0, 0 };
1631  vui64_t v64 = { 64, 64 };
1632 
1633  vt1 = vec_vclz ((vui64_t) vra);
1634  vt2 = (vui64_t) vec_cmplt(vt1, v64);
1635  vt3 = (vui64_t) vec_sld ((vui8_t) vzero, (vui8_t) vt2, 8);
1636  result = vec_andc (vt1, vt3);
1637  result = (vui64_t) vec_sums ((vi32_t) result, (vi32_t) vzero);
1638 #if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
1639  result = (vui64_t) vec_sld ((vui8_t) result, (vui8_t) result, 4);
1640 #endif
1641 #else
1642  /* vector clz instructions were introduced in power8. For power7 and
1643  * earlier, use the pveclib vec_clzw implementation. For a quadword
1644  * clz, this requires pre-conditioning the input before computing the
1645  * the word clz and sum across. */
1646  vui32_t c0, clz;
1647  vui32_t r32, gt32, gt32sr32, gt64sr64;
1648 
1649  c0 = vec_splat_u32 (0);
1650  gt32 = (vui32_t) vec_cmpgt ((vui32_t) vra, c0);
1651  gt32sr32 = vec_sld (c0, gt32, 12);
1652  gt64sr64 = vec_sld (c0, gt32, 8);
1653  gt32 = vec_sld (c0, gt32, 4);
1654 
1655  gt32sr32 = vec_or (gt32sr32, gt32);
1656  gt64sr64 = vec_or (gt64sr64, (vui32_t) vra);
1657  r32 = vec_or (gt32sr32, gt64sr64);
1658 
1659  clz = vec_clzw (r32);
1660  result = (vui64_t) vec_sums ((vi32_t) clz, (vi32_t) c0);
1661 #endif
1662 
1663  return ((vui128_t) result);
1664 }
