439 static inline vui128_t
440 vec_clzq (vui128_t vra)
441 {
442  vui64_t result;
443 
444 #ifdef _ARCH_PWR8
445  /*
446  * Use the Vector Count Leading Zeros Double Word instruction to get
447  * the count for the left and right vector halves. If the left vector
448  * doubleword of the input is nonzero then only the left count is
449  * included and we need to mask off the right count.
450  * Otherwise the left count is 64 and we need to add 64 to the right
451  * count.
452  * After masking we sum across the left and right counts to
453  * get the final 128-bit vector count (0-128).
454  */
455  vui64_t vt1, vt2, vt3;
456  vui64_t vzero = { 0, 0 };
457  vui64_t v64 = { 64, 64 };
458 
459  vt1 = vec_vclz ((vui64_t) vra);
460  vt2 = (vui64_t) vec_cmplt(vt1, v64);
461  vt3 = (vui64_t) vec_sld ((vui8_t) vzero, (vui8_t) vt2, 8);
462  result = vec_andc (vt1, vt3);
463 // print_vint128x ("vec_clzq and:", (vui128_t)result);
464  result = (vui64_t) vec_sums ((vi32_t) result, (vi32_t) vzero);
465 #if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
466  result = (vui64_t) vec_sld ((vui8_t) result, (vui8_t) result, 4);
467 #endif
468 // print_vint128x ("vec_clzq rtn:", (vui128_t)result);
469 #else
470  /* vector clz instructions were introduced in power8. For power7 and
471  * earlier, use the pveclib vec_clzw implementation. For a quadword
472  * clz, this requires pre-conditioning the input before computing the
473  * the word clz and sum across. */
474  vui32_t c0, clz;
475  vui32_t r32, gt32, gt32sr32, gt64sr64;
476 
477  c0 = vec_splat_u32 (0);
478  gt32 = (vui32_t)vec_cmpgt((vui32_t)vra, c0);
479  gt32sr32 = vec_sld (c0, gt32, 12);
480  gt64sr64 = vec_sld (c0, gt32, 8);
481  gt32 = vec_sld (c0, gt32, 4);
482 
483  gt32sr32 = vec_or (gt32sr32, gt32);
484  gt64sr64 = vec_or (gt64sr64, (vui32_t)vra);
485  r32 = vec_or (gt32sr32, gt64sr64);
486 
487  clz = vec_clzw (r32);
488  result = (vui64_t)vec_sums ((vi32_t)clz, (vi32_t)c0);
489 #endif
490 
491  return ((vui128_t) result);
492 }

504 static inline vui128_t
505 vec_cmul10ecuq (vui128_t *cout, vui128_t a, vui128_t cin)
506 {
507  vui32_t t;
508  vui32_t t_carry;
509 #ifdef _ARCH_PWR9
510  __asm__(
511  "vmul10ecuq %0,%2,%3;\n"
512  "vmul10euq %1,%2,%3;\n"
513  : "=v" (t_carry),
514  "=v" (t)
515  : "v" (a),
516  "v" (cin)
517  : );
518 #else
519  vui16_t ts = (vui16_t) a;
520  vui32_t tc;
521  vui16_t t10;
522  vui32_t t_odd, t_even, t_high;
523  vui32_t z = { 0, 0, 0, 0 };
524  t10 = vec_splat_u16(10);
525 #if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
526  t_even = vec_vmulouh (ts, t10);
527  t_odd = vec_vmuleuh (ts, t10);
528 #else
529  t_even = vec_vmuleuh(ts, t10);
530  t_odd = vec_vmulouh(ts, t10);
531 #endif
532  /* Shift t_even left 16-bits (right 112-bits) for the partial carry. */
533  t_high = vec_sld (z, t_even, 2);
534  /* Shift cin left 112 bits. */
535  tc = vec_sld ((vui32_t) cin, z, 14);
536  /* Shift t_even left 16 bits, merging the carry into the low bits. */
537  t_even = vec_sld (t_even, tc, 2);
538  /* then add the even/odd sub-products to generate the final product */
539 #ifdef _ARCH_PWR8
540  /* Any compiler that supports ARCH_PWR8 should support these builtins. */
541  t_carry = t_high; /* there is not carry into high */
542  t = (vui32_t) vec_vadduqm ((vui128_t) t_even, (vui128_t) t_odd);
543 #else
544  t_carry = t_high; /* there is no carry into high */
545  /* Use pveclib adduqm implementation for pre _ARCH_PWR8. */
546  t = (vui32_t) vec_adduqm ((vui128_t) t_even, (vui128_t) t_odd);
547 #endif
548 #endif
549  *cout = (vui128_t) t_carry;
550  return ((vui128_t) t);
551 }
