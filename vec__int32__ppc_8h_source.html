475 static inline vui32_t
476 vec_clzw (vui32_t vra)
477 {
478  vui32_t r;
479 #ifdef _ARCH_PWR8
480 #ifndef vec_vclzw
481  __asm__(
482  "vclzw %0,%1;"
483  : "=v" (r)
484  : "v" (vra)
485  : );
486 #else
487  r = vec_vclzw (vra);
488 #endif
489 #else
490 //#warning Implememention pre POWER8
491  vui32_t n, nt, y, x, s, m;
492  vui32_t z= {0,0,0,0};
493  vui32_t one = {1,1,1,1};
494 
495  /* n = 32 s = 16 */
496  s = vec_splat_u32(8);
497  s = vec_add (s, s);
498  n = vec_add (s, s);
499 
500  x = vra;
501  /* y=x>>16 if (y!=0) (n=n-16 x=y) */
502  y = vec_sr(x, s);
503  nt = vec_sub(n,s);
504  m = (vui32_t)vec_cmpgt(y, z);
505  s = vec_sr(s,one);
506  x = vec_sel (x, y, m);
507  n = vec_sel (n, nt, m);
508 
509  /* y=x>>8 if (y!=0) (n=n-8 x=y) */
510  y = vec_sr(x, s);
511  nt = vec_sub(n,s);
512  m = (vui32_t)vec_cmpgt(y, z);
513  s = vec_sr(s,one);
514  x = vec_sel (x, y, m);
515  n = vec_sel (n, nt, m);
516 
517  /* y=x>>4 if (y!=0) (n=n-4 x=y) */
518  y = vec_sr(x, s);
519  nt = vec_sub(n,s);
520  m = (vui32_t)vec_cmpgt(y, z);
521  s = vec_sr(s,one);
522  x = vec_sel (x, y, m);
523  n = vec_sel (n, nt, m);
524 
525  /* y=x>>2 if (y!=0) (n=n-2 x=y) */
526  y = vec_sr(x, s);
527  nt = vec_sub(n,s);
528  m = (vui32_t)vec_cmpgt(y, z);
529  s = vec_sr(s,one);
530  x = vec_sel (x, y, m);
531  n = vec_sel (n, nt, m);
532 
533  /* y=x>>1 if (y!=0) return (n=n-2) */
534  y = vec_sr(x, s);
535  nt = vec_sub(n,s);
536  nt = vec_sub(nt,s);
537  m = (vui32_t)vec_cmpgt(y, z);
538  n = vec_sel (n, nt, m);
539 
540  /* else return (x-n) */
541  nt = vec_sub (n, x);
542  n = vec_sel (nt, n, m);
543  r = n;
544 #endif
545  return ((vui32_t) r);
546 }

1210 static inline vui32_t
1211 vec_slwi (vui32_t vra, const unsigned int shb)
1212 {
1213  vui32_t lshift;
1214  vui32_t result;
1215 
1216  if (shb < 32)
1217  {
1218  /* Load the shift const in a vector. The element shifts require
1219  a shift amount for each element. For the immediate form the
1220  shift constant is splatted to all elements of the
1221  shift control. */
1222  if (__builtin_constant_p (shb) && (shb < 16))
1223  lshift = (vui32_t) vec_splat_s32(shb);
1224  else
1225  lshift = vec_splats ((unsigned int) shb);
1226 
1227  /* Vector Shift right bytes based on the lower 5-bits of
1228  corresponding element of lshift. */
1229  result = vec_vslw (vra, lshift);
1230  }
1231  else
1232  { /* shifts greater then 31 bits return zeros. */
1233  result = vec_xor ((vui32_t) vra, (vui32_t) vra);
1234  }
1235 
1236  return (vui32_t) result;
