251 static inline vui32_t
252 vec_clzw (vui32_t vra)
253 {
254  vui32_t r;
255 #ifdef _ARCH_PWR8
256 #ifndef vec_vclzw
257  __asm__(
258  "vclzw %0,%1;"
259  : "=v" (r)
260  : "v" (vra)
261  : );
262 #else
263  r = vec_vclzw (vra);
264 #endif
265 #else
266 //#warning Implememention pre POWER8
267  vui32_t n, nt, y, x, s, m;
268  vui32_t z= {0,0,0,0};
269  vui32_t one = {1,1,1,1};
270 
271  /* n = 32 s = 16 */
272  s = vec_splat_u32(8);
273  s = vec_add (s, s);
274  n = vec_add (s, s);
275 
276  x = vra;
277  /* y=x>>16 if (y!=0) (n=n-16 x=y) */
278  y = vec_sr(x, s);
279  nt = vec_sub(n,s);
280  m = (vui32_t)vec_cmpgt(y, z);
281  s = vec_sr(s,one);
282  x = vec_sel (x, y, m);
283  n = vec_sel (n, nt, m);
284 
285  /* y=x>>8 if (y!=0) (n=n-8 x=y) */
286  y = vec_sr(x, s);
287  nt = vec_sub(n,s);
288  m = (vui32_t)vec_cmpgt(y, z);
289  s = vec_sr(s,one);
290  x = vec_sel (x, y, m);
291  n = vec_sel (n, nt, m);
292 
293  /* y=x>>4 if (y!=0) (n=n-4 x=y) */
294  y = vec_sr(x, s);
295  nt = vec_sub(n,s);
296  m = (vui32_t)vec_cmpgt(y, z);
297  s = vec_sr(s,one);
298  x = vec_sel (x, y, m);
299  n = vec_sel (n, nt, m);
300 
301  /* y=x>>2 if (y!=0) (n=n-2 x=y) */
302  y = vec_sr(x, s);
303  nt = vec_sub(n,s);
304  m = (vui32_t)vec_cmpgt(y, z);
305  s = vec_sr(s,one);
306  x = vec_sel (x, y, m);
307  n = vec_sel (n, nt, m);
308 
309  /* y=x>>1 if (y!=0) return (n=n-2) */
310  y = vec_sr(x, s);
311  nt = vec_sub(n,s);
312  nt = vec_sub(nt,s);
313  m = (vui32_t)vec_cmpgt(y, z);
314  n = vec_sel (n, nt, m);
315 
316  /* else return (x-n) */
317  nt = vec_sub (n, x);
318  n = vec_sel (nt, n, m);
319  r = n;
320 #endif
321  return ((vui32_t) r);

600 static inline vui64_t
601 vec_muleuw (vui32_t a, vui32_t b)
602 {
603  vui64_t res;
604 #ifdef _ARCH_PWR8
605 #if defined __GNUC__ && (__GNUC__ > 7)
606  res = vec_mule (a, b);
607 #else
608  __asm__(
609 #if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
610  "vmulouw %0,%1,%2;\n"
611 #else
612  "vmuleuw %0,%1,%2;\n"
613 #endif
614  : "=v" (res)
615  : "v" (a),
616  "v" (b)
617  : );
618 #endif
619 #else
620  const vui32_t zero = {0,0,0,0};
621  const vui32_t ones = {-1,-1,-1,-1};
622  vui32_t wmask01;
623  vui32_t p0, p1, pp10, pp01, resw;
624  vui16_t m0, m1, mt, mth, mtl;
625 
626  /* generate {0,-1,0,-1} mask. */
627  wmask01 = vec_vmrghw (zero, ones);
628 
629  mt = (vui16_t)b;
630  mtl = vec_mergeh (mt, mt);
631  mth = vec_mergel (mt, mt);
632 
633 #ifdef _ARCH_PWR7
634  m0 = (vui16_t)vec_xxpermdi ((vui64_t)mtl, (vui64_t)mth, 0);
635 #else
636  {
637  vui32_t temp;
638  temp = vec_sld ((vui32_t) mtl, (vui32_t) mth, 8);
639  m0 = (vui16_t) vec_sld (temp, (vui32_t) mth, 8);
640  }
641 #endif
642 
643  resw = vec_sld (a, a, 12);
644  resw = vec_sel (a, resw, wmask01);
645  m1 = (vui16_t)resw;
646 
647  p0 = vec_vmuleuh (m1, m0);
648  p1 = vec_vmulouh (m1, m0);
649  resw = vec_sel (p0, p1, wmask01);
650  res = (vui64_t)resw;
651 
652  pp10 = vec_sld (p1, p1, 12);
653  pp01 = p0;
654  /* pp01 = vec_addudm (pp01, pp10). */
655  {
656  vui32_t c;
657  vui32_t xmask;
658  xmask = vec_sld (wmask01, wmask01, 2);
659  c = vec_vaddcuw (pp01, pp10);
660  pp01 = vec_vadduwm (pp01, pp10);
661  c = vec_sld (c, c, 6);
662  pp01 = vec_sld (pp01, pp01, 2);
663  pp01 = vec_sel (c, pp01, xmask);
664  }
665  /* res = vec_addudm (pp01, res). */
666  {
667  vui32_t c, r;
668  c = vec_vaddcuw (pp01, (vui32_t)res);
669  r = vec_vadduwm (pp01, (vui32_t)res);
670  c = vec_sld (c, zero, 4);
671  res = (vui64_t)vec_vadduwm (r, c);
672  }
673 #endif

983 static inline vui32_t
984 vec_slwi (vui32_t vra, const unsigned int shb)
985 {
986  vui32_t lshift;
987  vui32_t result;
988 
989  if (shb < 32)
990  {
991  /* Load the shift const in a vector. The element shifts require
992  a shift amount for each element. For the immediate form the
993  shift constant is splatted to all elements of the
994  shift control. */
995  if (__builtin_constant_p (shb) && (shb < 16))
996  lshift = (vui32_t) vec_splat_s32(shb);
997  else
998  lshift = vec_splats ((unsigned int) shb);
999 
1000  /* Vector Shift right bytes based on the lower 5-bits of
1001  corresponding element of lshift. */
1002  result = vec_vslw (vra, lshift);
1003  }
1004  else
1005  { /* shifts greater then 31 bits return zeros. */
1006  result = vec_xor ((vui32_t) vra, (vui32_t) vra);
1007  }
1008 
1009  return (vui32_t) result;
