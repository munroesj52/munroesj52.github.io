462 #endif
463 #else

465  vui32_t n, nt, y, x, s, m;
466  vui32_t z= {0,0,0,0};
467  vui32_t one = {1,1,1,1};

469  /* n = 32 s = 16 */
470  s = vec_splat_u32(8);
471  s = vec_add (s, s);
472  n = vec_add (s, s);

474  x = vra;

481  n = vec_sel (n, nt, m);

484  y = vec_sr(x, s);
485  nt = vec_sub(n,s);
486  m = (vui32_t)vec_cmpgt(y, z);
487  s = vec_sr(s,one);
488  x = vec_sel (x, y, m);
489  n = vec_sel (n, nt, m);
490 
491  /* y=x>>4 if (y!=0) (n=n-4 x=y) */
492  y = vec_sr(x, s);
493  nt = vec_sub(n,s);
494  m = (vui32_t)vec_cmpgt(y, z);
495  s = vec_sr(s,one);
496  x = vec_sel (x, y, m);
497  n = vec_sel (n, nt, m);
498 
499  /* y=x>>2 if (y!=0) (n=n-2 x=y) */
500  y = vec_sr(x, s);
501  nt = vec_sub(n,s);
502  m = (vui32_t)vec_cmpgt(y, z);
503  s = vec_sr(s,one);
504  x = vec_sel (x, y, m);
505  n = vec_sel (n, nt, m);
506 
507  /* y=x>>1 if (y!=0) return (n=n-2) */
508  y = vec_sr(x, s);
509  nt = vec_sub(n,s);
510  nt = vec_sub(nt,s);
511  m = (vui32_t)vec_cmpgt(y, z);
512  n = vec_sel (n, nt, m);
513

518 #endif

1188 {
1189  vui32_t lshift;
1190  vui32_t result;
1191 
1192  if (shb < 32)
1193  {
1194  /* Load the shift const in a vector. The element shifts require
1195  a shift amount for each element. For the immediate form the
1196  shift constant is splatted to all elements of the
1197  shift control. */
1198  if (__builtin_constant_p (shb) && (shb < 16))
1199  lshift = (vui32_t) vec_splat_s32(shb);
1200  else
1201  lshift = vec_splats ((unsigned int) shb);
1202 
1203  /* Vector Shift right bytes based on the lower 5-bits of
1204  corresponding element of lshift. */
1205  result = vec_vslw (vra, lshift);
1206  }
1207  else
1208  { /* shifts greater then 31 bits return zeros. */
1209  result = vec_xor ((vui32_t) vra, (vui32_t) vra);
1210  }
